## Game Concept

### One-Liner
> A point-and-click historical game where you advise Rome through 5 pivotal moments, but your advisors have hidden agendas and may be manipulating you.

### Core Loop
1. Enter a historical Roman scene
2. Examine evidence (changes your stats based on what evidence suggests)
3. Talk to advisors (conversation length affects stats)
4. Make a pivotal decision
5. See how your choice changed things (not what actually happened)
6. Proceed to next era with updated stats
7. At the end, compare your timeline to actual history

### The Hook
- Advisors are LLM-powered and will lie to serve their faction
- Evidence can reveal their lies
- Your stats reflect YOUR version of Rome, not real Rome
- Only at the very end do you see how your Rome compares to actual history

#### 2. Advisor Conversations (On End)

Stats change when conversation ends, based on depth:

```python
ADVISOR_BASE_INFLUENCE = {
    "caelius": {        # Senator - pushes Republic, fears Military
        "republic": +5,
        "military": -3,
        "economy": +2
    },
    "labienus": {       # General - pushes Military
        "military": +5,
        "stability": +3,
        "republic": -3
    },
    "balbus": {         # Merchant - pushes Economy
        "economy": +5,
        "stability": +2,
        "military": -2
    },
    "metellus": {       # Priest - pushes Stability/Tradition
        "stability": +5,
        "republic": +3,
        "economy": -2
    }
}

CONVERSATION_MULTIPLIERS = {
    0: 0,       # Opened and closed immediately - no influence
    1: 0.5,    # 1-2 messages - minor influence
    2: 0.5,
    3: 1.0,    # 3-5 messages - full influence
    4: 1.0,
    5: 1.0,
    6: 1.5,    # 6+ messages - strong influence (you listened a lot)
}

def calculate_conversation_impact(advisor_id: str, message_count: int) -> dict:
    base = ADVISOR_BASE_INFLUENCE[advisor_id]
    multiplier = CONVERSATION_MULTIPLIERS.get(message_count, 1.5)  # 6+ = 1.5
    
    return {
        stat: int(value * multiplier)
        for stat, value in base.items()
    }
```

**Behavior:**
- Player chats with advisor
- Player clicks "End Conversation"
- Message count determines multiplier
- Stats update based on advisor's faction Ã— multiplier

### Stat Bounds

Stats are always clamped between 0-100:

```python
def update_stat(current: int, change: int) -> int:
    return max(0, min(100, current + change))
```

### Correlation Score Calculation

At the end of the game, calculate how closely the player's choices matched history:

```python
HISTORICAL_CHOICES = {
    "era1": "A",  # Resist
    "era2": "A",  # Assassinate
    "era3": "A",  # Accept Principate
    "era4": "A",  # Centralize
    "era5": "A",  # Resist barbarians
}

def calculate_correlation(player_choices: dict) -> int:
    matches = sum(
        1 for era, choice in player_choices.items()
        if choice == HISTORICAL_CHOICES.get(era)
    )
    return int((matches / 5) * 100)  # 0-100%
```

```typescript
// frontend/src/store/gameStore.ts

import { create } from 'zustand'

interface Stats {
  military: number
  economy: number
  stability: number
  republic: number
}

interface StatChange {
  stat: keyof Stats
  oldValue: number
  newValue: number
  change: number
}

interface GameState {
  // Session
  sessionId: string | null
  currentEra: number
  stats: Stats
  
  // Current era state
  eraData: EraData | null
  evidenceViewed: string[]
  advisorsChatted: string[]
  currentChat: { advisorId: string; messages: ChatMessage[] } | null
  
  // History
  choices: Record<string, string>
  
  // UI State
  activePopup: 'evidence' | 'dialogue' | null
  activeItem: string | null
  pendingStatChanges: StatChange[]
  
  // Actions
  startGame: () => Promise<void>
  loadEra: (eraId: number) => Promise<void>
  viewEvidence: (evidenceId: string) => Promise<void>
  startChat: (advisorId: string) => void
  sendMessage: (message: string) => Promise<void>
  endChat: () => Promise<void>
  makeDecision: (choice: string) => Promise<void>
  closePopup: () => void
}

export const useGameStore = create<GameState>((set, get) => ({
  // Initial state
  sessionId: null,
  currentEra: 1,
  stats: { military: 50, economy: 50, stability: 50, republic: 50 },
  eraData: null,
  evidenceViewed: [],
  advisorsChatted: [],
  currentChat: null,
  choices: {},
  activePopup: null,
  activeItem: null,
  pendingStatChanges: [],
  
  startGame: async () => {
    const res = await fetch('/api/game/start', { method: 'POST' })
    const data = await res.json()
    set({
      sessionId: data.session_id,
      stats: data.stats,
      currentEra: 1
    })
  },
  
  loadEra: async (eraId: number) => {
    const { sessionId } = get()
    const res = await fetch(`/api/era/${eraId}?session_id=${sessionId}`)
    const data = await res.json()
    set({
      eraData: data,
      currentEra: eraId,
      evidenceViewed: [],
      advisorsChatted: [],
      currentChat: null
    })
  },
  
  viewEvidence: async (evidenceId: string) => {
    const { sessionId, evidenceViewed } = get()
    
    // Don't re-fetch if already viewed
    if (evidenceViewed.includes(evidenceId)) {
      set({ activePopup: 'evidence', activeItem: evidenceId })
      return
    }
    
    const res = await fetch(`/api/evidence/${evidenceId}/view`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ session_id: sessionId })
    })
    const data = await res.json()
    
    set({
      activePopup: 'evidence',
      activeItem: evidenceId,
      evidenceViewed: [...evidenceViewed, evidenceId],
      stats: data.new_stats,
      pendingStatChanges: data.stat_changes
    })
  },
  
  startChat: (advisorId: string) => {
    set({
      activePopup: 'dialogue',
      activeItem: advisorId,
      currentChat: { advisorId, messages: [] }
    })
  },
  
  sendMessage: async (message: string) => {
    const { sessionId, currentChat } = get()
    if (!currentChat) return
    
    // Add user message optimistically
    const userMessage = { role: 'user', content: message }
    set({
      currentChat: {
        ...currentChat,
        messages: [...currentChat.messages, userMessage]
      }
    })
    
    // Get AI response
    const res = await fetch(`/api/advisor/${currentChat.advisorId}/chat`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ session_id: sessionId, message })
    })
    const data = await res.json()
    
    // Add AI response
    const aiMessage = { role: 'assistant', content: data.response }
    set(state => ({
      currentChat: state.currentChat ? {
        ...state.currentChat,
        messages: [...state.currentChat.messages, aiMessage]
      } : null
    }))
  },
  
  endChat: async () => {
    const { sessionId, currentChat, advisorsChatted } = get()
    if (!currentChat) return
    
    const res = await fetch(`/api/advisor/${currentChat.advisorId}/end`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ session_id: sessionId })
    })
    const data = await res.json()
    
    set({
      activePopup: null,
      activeItem: null,
      currentChat: null,
      advisorsChatted: [...advisorsChatted, currentChat.advisorId],
      stats: data.new_stats,
      pendingStatChanges: data.stat_changes
    })
  },
  
  makeDecision: async (choice: string) => {
    const { sessionId, currentEra, choices } = get()
    
    const res = await fetch('/api/decision', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({
        session_id: sessionId,
        era_id: currentEra,
        choice
      })
    })
    const data = await res.json()
    
    set({
      choices: { ...choices, [`era${currentEra}`]: choice },
      stats: data.new_stats,
      pendingStatChanges: data.stat_changes
    })
    
    return data  // Return for navigation
  },
  
  closePopup: () => {
    set({
      activePopup: null,
      activeItem: null,
      pendingStatChanges: []
    })
  }
}))

def build_advisor_prompt(advisor: AdvisorConfig, era_context: str, chat_history: list) -> str:
    return f"""You are {advisor.name}, {advisor.title} in ancient Rome.

## YOUR CHARACTER
Speaking style: {advisor.speaking_style}
Background: {advisor.background}

## YOUR PUBLIC POSITION (what you claim)
{advisor.public_position}

## YOUR HIDDEN AGENDA (never reveal directly)
{advisor.hidden_agenda}

## WHAT YOU KNOW
{chr(10).join(f"- {k}" for k in advisor.knowledge)}

## TOPICS YOU WILL LIE ABOUT OR AVOID
{chr(10).join(f"- {lie}" for lie in advisor.will_lie_about)}

## YOUR "TELLS" (subtle signs when you're being deceptive)
{chr(10).join(f"- {tell}" for tell in advisor.tells)}

## CURRENT SITUATION
{era_context}

## CONVERSATION RULES
1. Stay in character as a Roman of this era
2. Advocate for your faction's interests
3. You may lie or deflect about topics in your "will lie about" list
4. If pressed hard on something you're lying about, show subtle discomfort (one of your "tells")
5. NEVER explicitly reveal your hidden agenda
6. Keep responses to 2-4 sentences unless asked for detail
7. Use period-appropriate language and references
8. If asked something outside your knowledge, admit you don't know

## CONVERSATION SO FAR
{format_history(chat_history)}

Respond as {advisor.name}. Stay in character."""

def format_history(history: list) -> str:
    if not history:
        return "[This is the start of the conversation]"
    
    lines = []
    for msg in history[-10:]:  # Last 10 messages
        speaker = "Oracle (Player)" if msg["role"] == "user" else "You"
        lines.append(f"{speaker}: {msg['content']}")
    return "\n".join(lines)
```
